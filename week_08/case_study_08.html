library(tidyverse)

URL<- "https://gml.noaa.gov/webdata/ccgg/trends/co2/co2_annmean_mlo.txt"
Data<- read_table(URL, skip=57, col_names=c("year", "mean", "unc"))
## 
## -- Column specification ------------------------------------------------------
## cols(
##   year = col_double(),
##   mean = col_double(),
##   unc = col_double()
## )
ggplot(data=Data, aes(x=year, y=mean))+
  geom_line()

top5 <- Data %>%
  arrange(desc(mean))%>%
  top_n(mean, n=5)

knitr::kable(top5)
